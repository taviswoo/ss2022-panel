@app.route("/login", methods=["GET", "POST"]) def login(): cfg = load_config() # 如果未初始化，跳 setup if "ADMIN_USER" not in cfg or "ADMIN_PASSWORD" not in cfg: return redirect(url_for("setup")) if request.method == "POST": user = request.form["username"] pwd = request.form["password"] if user == cfg["ADMIN_USER"] and pwd == cfg["ADMIN_PASSWORD"]: session["logged_in"] = True return redirect(url_for("index")) return "账号或密码错误" return render_template("login.html")
